# 工具类的使用：

  1. 在grade.properties 中添加如下代码
FVL_TOOL_RELEASE_URL=http://172.16.13.125:8081/repository/fvl_tools_release/
FVL_TOOL_SNAPSHOTS_URL=http://172.16.13.125:8081/repository/fvl_tools_snapshot/

POM_GROUPID=com.fvlprojects.android
POM_PACKAGING=aar

NEXUS_USERNAME=admin
NEXUS_PASSWORD=admin
